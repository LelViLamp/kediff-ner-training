Do we find some Zipf-iness in the tokenised text?

token_counter <-
  here::here("data", "token_counter.csv") %>%
  read.csv(col.names = c("Token", "Count")) %>%
  mutate(Token = factor(Token, levels = Token)) %>%
  arrange(-Count)
token_counter
token_counter %>%
  ggplot(aes(x = Token, y = Count)) +
  geom_bar(stat = "identity")
